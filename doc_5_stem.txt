#includ <stdio.h>
#includ <stdlib.h>
#includ <ctype.h>
int main(int argc, char *argv[])
{
int letter[26];
file *fin;
int i;
char onechar;
for ( i = 0; i < 26; i++ ) letter[i] = 0;
fin = fopen (argv[1], "r");
if (fin == null )
{
printf ("unabl to open file %s\n", argv[1] );
exit(1);
}
while ( (onechar = getc(fin) ) != eof )
{
if ( isalpha(onechar) ) ++letter[tolow(onechar) - 'a'];
}
fclose (fin);
for ( i = 0; i < 26; i++ )
{
printf ("letter %c occur %d time\n", i + 'a', letter[i]);
}
}
